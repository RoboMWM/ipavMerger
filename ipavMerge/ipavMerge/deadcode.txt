        {
            int key = 0;

            {

                while (line.Substring(0,1).Equals("-"))
                {
                    String[] stupid = line.Split('\"');
                    key++;
                    Console.WriteLine(stupid.Length + " key: " + key);
                    child.Add(new Child(stupid[0], 
                        stupid[1],
                        stupid[2],
                        stupid[3],
                        stupid[4],
                        stupid[5],
                        stupid[6],
                        stupid[7]));
                    line = file.ReadLine();
                    if (line == null)
                        break;
                }
                leList.Add(new Parent(parent, child));
            }
            return leList;
        }
